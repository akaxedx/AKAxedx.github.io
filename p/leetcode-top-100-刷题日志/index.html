class Solution {
    public int[] twoSum(int[] nums, int target) {
        Map<Integer, Integer> hashtable = new HashMap<Integer, Integer>();
        for (int i = 0; i < nums.length; ++i) {
            if (hashtable.containsKey(target - nums[i])) {
                return new int[]{hashtable.get(target - nums[i]), i};
            }
            hashtable.put(nums[i], i);
        }
        return new int[0];
    }
}

class Solution {
    public List<List<String>> groupAnagrams(String[] strs) {
        Map<String, List<String>> map = new HashMap<>();
        for (int i = 0; i < strs.length; i++) {
            char[] s = strs[i].toCharArray();
            Arrays.sort(s);
            map.computeIfAbsent(new String(s), k->new ArrayList<>()).add(strs[i]);
        }
        return new ArrayList<>(map.values());
    }
}

class Solution {
    public int longestConsecutive(int[] nums) {
        if(nums.length == 0) {
            return 0;
        }
       Arrays.sort(nums);
        int left = 0;
        int result = 1;
        int currentNum = nums[left];
        for (int i = 1; i < nums.length; i++ ) {
            if (nums[i] != currentNum && nums[i] != currentNum+1) {
                left = i;
                currentNum = nums[left];
            } else {
                currentNum = nums[i];
                result = Math.max(result, currentNum - nums[left] + 1);
            }

class Solution {
    public void moveZeroes(int[] nums) {
        int n = nums.length, left = 0, right = 0;
        while (right < n) {
            if (nums[right] != 0) {
                swap(nums, left, right);
                left++;
            }
            right++;
        }
    }

    public void swap(int[] nums, int left, int right) {
        int temp = nums[left];
        nums[left] = nums[right];
        nums[right] = temp;
    }
}

class Solution {
    public int maxArea(int[] height) {
        int result = 0;
        int left = 0;
        int right = height.length-1;
        while (left != right) {
            result =Math.max(Math.min(height[left],height[right]) * (right - left), result);
            if (height[left] >= height[right]) {
                right--;
            } else{
                left ++;
            }
        }
        return result;
    }
}

class Solution {
    public List<List<Integer>> threeSum(int[] nums) {
        Arrays.sort(nums);
        List<List<Integer>> result = new ArrayList<>();
        int last = nums[0] - 1;
        for (int i = 0; i<nums.length - 2;i++) {
            int left = i+1;
            int right = nums.length-1;
            if (nums[i] == last) {
                continue;
            }
            if (nums[i] > 0 ) return result;
            if (nums[right] < 0) return result;
            last = nums[i];
            while (left < right) {
                int sum = nums[i] + nums[left] + nums[right];
                if(sum == 0) {
                    result.add(List.of(nums[i],nums[left],nums[right]));
                    for (left++;left<right && nums[left] == nums[left-1];left++);
                    for (right--;left<right && nums[right] == nums[right+1];right--);
                } else if (sum >0) {
                         right --;
                } else {
                    left++;
                }
            }
            
        }

        return result;
    }
}

class Solution {
    public int trap(int[] height) {
        int[] Hl = new int[height.length];
        int[] Hr = new int[height.length];
        // Â∑¶Âè≥ÊùøÈ´òÊï∞ÁªÑ
        int j = height.length - 1;
        int hl = 0;
        int hr = 0;
        int result = 0;
        for (int i = 0; i < Hl.length; i++) {
            hl = Math.max(hl,height[i]);
            hr = Math.max(hr,height[j]);
            Hl[i] = hl;
            Hr[j] = hr;
            j--;
        }
        for (int i = 0; i < Hr.length; i++) {
            result += Math.min(Hl[i],Hr[i]) - height[i];
        }
        return result;
    }
}

class Solution {
    public int trap(int[] height) {
        int ans = 0;
        int left = 0;
        int right = height.length - 1;
        int preMax = 0; // ÂâçÁºÄÊúÄÂ§ßÂÄºÔºåÈöèÁùÄÂ∑¶ÊåáÈíà left ÁöÑÁßªÂä®ËÄåÊõ¥Êñ∞

        while (left < right) {
            preMax = Math.max(preMax, height[left]);

        return ans;
    }

class Solution {
    public int lengthOfLongestSubstring(String s) {
        char[] cs = s.toCharArray();
        Map<Character, Boolean> map = new HashMap<>();
        int left = 0;
        int right = 0;
        int result = 0;
        for (char c : cs) {
            if (!map.containsKey(c)) {
                map.put(c, true);
                right++;
            } else break;
        }
        if (right == cs.length) {
            return right;
        }
        right --;
        while (right <= cs.length - 1) {
            map.clear();
            for (int i = left; i<= right && right <=cs.length-1;i++){
                if (!map.containsKey(cs[i])) {
                    map.put(cs[i],true);
                    if (i == right) {
                        right++;
                    }
                    if (right == cs.length) {
                        right --;
                        return Math.max(result, right - left + 1);
                    }
                } else {
                    if (i == right) {
                        left ++;
                    }
                    result = Math.max(result, right - left + 1);
                    break;
                }
            }
        }
        return Math.max(result, right - left + 1);
    }
}

class Solution {
    public int lengthOfLongestSubstring(String s) {
       Map<Character, Integer> m = new HashMap<>();
        char[] cs = s.toCharArray();
        int left = 0;
        int result = 0;
        for (int right = 0; right < cs.length; right++) {
            if (!m.containsKey(cs[right])) {
                m.put(cs[right], 0);
            }
            m.put(cs[right], m.get(cs[right])+ 1);
            while (m.get(cs[right]) > 1) {
                m.put(cs[left], m.get(cs[left]) - 1);
                left++;
            }
            result = Math.max(result, right - left + 1);
        }
        return result;
    }
}
